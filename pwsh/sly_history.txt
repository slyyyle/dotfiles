$env:STARSHIP_CONFIG = "$env:USERPROFILE/.config/starship.toml"
# Ensure PATH includes only necessary directories
$env:PATH = "$env:USERPROFILE/shortcuts;" + $env:PATH
# Initialize right prompt variable
$global:rightPrompt = ""
# Improve Terminal Integration with Starship PreCommand Hook
function Invoke-Starship-PreCommand {`
    $current_location = $executionContext.SessionState.Path.CurrentLocation`
    if ($current_location.Provider.Name -eq "FileSystem") {`
        $ansi_escape = [char]27`
        $provider_path = $current_location.ProviderPath -replace "\\", "/"`
        $prompt = "$ansi_escape]7;file://${env:COMPUTERNAME}/${provider_path}$ansi_escape\"`
    }`
    $host.ui.Write($prompt)`
    `
    # Update right prompt content - customize this as needed`
    $global:rightPrompt = Get-Date -Format "HH:mm:ss"`
}
# Custom prompt function that works with Starship
function prompt {`
    # Get the Starship prompt`
    $standardPrompt = (& starship prompt)`
    `
    # Calculate the position for the right prompt`
    $cursorPos = $host.UI.RawUI.CursorPosition`
    $windowWidth = $host.UI.RawUI.WindowSize.Width`
    $rightPromptLength = $global:rightPrompt.Length`
    `
    # Position and write the right prompt`
    $spacesToRight = $windowWidth - $cursorPos.X - $rightPromptLength - 1`
    if ($spacesToRight -gt 0) {`
        Write-Host (" " * $spacesToRight) -NoNewline`
        Write-Host $global:rightPrompt -ForegroundColor DarkGray -NoNewline`
